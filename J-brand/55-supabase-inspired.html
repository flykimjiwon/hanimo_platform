바로 시작하세요

$ curl -L https://hanimo.dev/install/hanimo-darwin-arm64 -o hanimo
$ chmod +x hanimo && mv hanimo /usr/local/bin/
$ hanimo
# macOS / Linux via Homebrew
$ brew install hanimo
$ hanimo
# Build from source (requires Go 1.22+)
$ git clone https://github.com/flykimjiwon/hanimo
$ cd hanimo && go build ./cmd/hanimo
$ ./hanimo
